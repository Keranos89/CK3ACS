acs_init_global_flag = {
    effect = {
        set_global_variable = { name = mod_acs_active value = yes }
    }
}

acs_window = {
    scope = character
    
    is_shown = { 
        NOT = { has_variable = is_acs_building_list }
    }

    effect = {
        create_searched_character_list = yes
    }
}

close_acs_window = {
    scope = character
    
    effect = {
        clear_acs_list = yes
    }
}

acs_reset_filters_and_sorting = {
    effect = {
        acs_reset_filters_and_sorting = yes
        acs_auto_apply_sorting_and_filters = yes
    }
}

acs_auto_apply_sorting_and_filters = {
    scope = character
         
    is_shown = { 
        has_global_variable = acs_auto_apply_sorting_and_filters
    }
        
    effect = {
        if = {
            limit = { has_global_variable = acs_auto_apply_sorting_and_filters }
            remove_global_variable = acs_auto_apply_sorting_and_filters
        }
        else = {
            set_global_variable = acs_auto_apply_sorting_and_filters
        }
        acs_auto_apply_sorting_and_filters = yes
    }
}

acs_undo_filters = {
    effect = {
        acs_load_undo_0_filters = yes
    }
}
