	AND = {
		OR = {
			AND = {
                OR = {
					NOT = {  has_global_variable = acs_filter_trait_personality_lustful }
					global_var:acs_filter_trait_personality_lustful = no
				}
				OR = {
					NOT = {  has_global_variable = acs_filter_trait_personality_chaste }
					global_var:acs_filter_trait_personality_chaste = no
				}
			}
            AND = {
			    global_var:acs_filter_trait_personality_lustful = yes
