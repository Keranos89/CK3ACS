acs_filter_trait_health_wounded = {
	AND = {
		OR = {
			AND = {
                OR = {
					NOT = {  has_global_variable = acs_filter_trait_health_wounded_1 }
					global_var:acs_filter_trait_health_wounded_1 = no
				}
				OR = {
					NOT = {  has_global_variable = acs_filter_trait_health_wounded_2 }
					global_var:acs_filter_trait_health_wounded_2 = no
				}
				OR = {
					NOT = {  has_global_variable = acs_filter_trait_health_wounded_3 }
					global_var:acs_filter_trait_health_wounded_3 = no
				}
			}
            AND = {
			    global_var:acs_filter_trait_health_wounded_1 = yes
			    has_trait = wounded_1
			}
			AND = {
			    global_var:acs_filter_trait_health_wounded_2 = yes
			    has_trait = wounded_2
			}
			AND = {
			    global_var:acs_filter_trait_health_wounded_3 = yes
			    has_trait = wounded_3
			}
		}
		NOR = {
            AND = {
			    global_var:acs_filter_trait_health_wounded_1 = no
			    has_trait = wounded_1
			}
			AND = {
			    global_var:acs_filter_trait_health_wounded_2 = no
			    has_trait = wounded_2
			}
			AND = {
			    global_var:acs_filter_trait_health_wounded_3 = no
			    has_trait = wounded_3
			}
		}
	}
}
