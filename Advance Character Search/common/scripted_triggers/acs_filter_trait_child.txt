acs_filter_trait_child_rowdy = {
	OR = {
		AND = {
		    global_var:acs_filter_trait_child_rowdy = yes
		    has_trait = rowdy
		}
		AND = {
		    global_var:acs_filter_trait_child_rowdy = no
		    has_trait = rowdy
		}
	}
}
