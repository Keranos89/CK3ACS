	AND = {
		OR = {
			AND = {
                OR = {

					global_var:acs_filter_trait_fame_born_status_bastard = no
				}
				OR = {
					NOT = {  has_global_variable = acs_filter_trait_fame_born_status_legitimized_bastard }
					global_var:acs_filter_trait_fame_born_status_legitimized_bastard = no

					NOT = {  has_global_variable = acs_filter_trait_fame_born_status_disputed_heritage }
					global_var:acs_filter_trait_fame_born_status_disputed_heritage = no
				}
				OR = {
					NOT = {  has_global_variable = acs_filter_trait_fame_born_status_child_of_concubine_female }
					global_var:acs_filter_trait_fame_born_status_child_of_concubine_female = no
				}
				OR = {
					NOT = {  has_global_variable = acs_filter_trait_fame_born_status_child_of_concubine_male }
					global_var:acs_filter_trait_fame_born_status_child_of_concubine_male = no
				}
				OR = {
					NOT = {  has_global_variable = acs_filter_trait_fame_born_status_wild_oat }
					global_var:acs_filter_trait_fame_born_status_wild_oat = no
				}
				OR = {
					NOT = {  has_global_variable = acs_filter_trait_fame_born_status_bastard_founder }
					global_var:acs_filter_trait_fame_born_status_bastard_founder = no
				}
			}
            AND = {
			    global_var:acs_filter_trait_fame_born_status_bastard = yes
			    has_trait = bastard
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_legitimized_bastard = yes
			    has_trait = legitimized_bastard
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_disputed_heritage = yes
			    has_trait = disputed_heritage
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_child_of_concubine_female = yes
			    has_trait = child_of_concubine_female
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_child_of_concubine_male = yes
			    has_trait = child_of_concubine_male
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_wild_oat = yes
			    has_trait = wild_oat
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_bastard_founder = yes
			    has_trait = bastard_founder
			}
		}
		NOR = {
            AND = {
			    global_var:acs_filter_trait_fame_born_status_bastard = no
			    has_trait = bastard
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_legitimized_bastard = no
			    has_trait = legitimized_bastard
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_disputed_heritage = no
			    has_trait = disputed_heritage
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_child_of_concubine_female = no
			    has_trait = child_of_concubine_female
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_child_of_concubine_male = no
			    has_trait = child_of_concubine_male
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_wild_oat = no
			    has_trait = wild_oat
			}
			AND = {
			    global_var:acs_filter_trait_fame_born_status_bastard_founder = no
			    has_trait = bastard_founder
			}
		}
	}
}
