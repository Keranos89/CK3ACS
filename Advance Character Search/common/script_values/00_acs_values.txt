claim_value = {
    value = 0
    if = {
        limit = { tier < tier_duchy }
        add = 1
    }
    if = {
        limit = { tier = tier_duchy }
        add = 10
    }
    if = {
        limit = { tier = tier_kingdom }
        add = 100
    }
    if = {
        limit = { tier = tier_empire }

claim_value_for_sort = {
    value = claim_value
    multiply = 10

        limit = {

    }
    if = {
        limit = {
            NOT = {
                title_capital_county.culture_group = prev.culture_group

        add = -2
    }
}
