    is_shown = {
		AND = {
			global_var:acs_filter_trait_fame_kinslayer_1 = yes
			global_var:acs_filter_trait_fame_kinslayer_2 = yes
			global_var:acs_filter_trait_fame_kinslayer_3 = yes
		} 
    } 
        
    effect = {
        if = {
            limit = { 
				NOR = {
					has_global_variable = acs_filter_trait_fame_kinslayer_1
					has_global_variable = acs_filter_trait_fame_kinslayer_2
					has_global_variable = acs_filter_trait_fame_kinslayer_3
				}
			}
			set_global_variable = {  name = acs_filter_trait_fame_kinslayer_1 value = yes }
			set_global_variable = {  name = acs_filter_trait_fame_kinslayer_2 value = yes }
			set_global_variable = {  name = acs_filter_trait_fame_kinslayer_3 value = yes }
        }
        else = {
            if = {
                limit = {
					AND = {
						global_var:acs_filter_trait_fame_kinslayer_1 = yes
						global_var:acs_filter_trait_fame_kinslayer_2 = yes
						global_var:acs_filter_trait_fame_kinslayer_3 = yes
					}
				}
				set_global_variable = {  name = acs_filter_trait_fame_kinslayer_1 value = no }
				set_global_variable = {  name = acs_filter_trait_fame_kinslayer_2 value = no }
				set_global_variable = {  name = acs_filter_trait_fame_kinslayer_3 value = no }
            }
            else = {
				remove_global_variable = acs_filter_trait_fame_kinslayer_1
			remove_global_variable = acs_filter_trait_fame_kinslayer_2
			remove_global_variable = acs_filter_trait_fame_kinslayer_3
            }

        }
        acs_auto_apply_sorting_and_filters = yes
    }
