acs_filter_trait_religius_celibate_negative = {
    is_shown = {
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:celibate_negative = this
        }
    }
}

acs_filter_trait_religius_celibate = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:celibate = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        OR = {
                            flag:celibate = this
                            flag:celibate_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_active_filter_list target = flag:celibate }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        flag:celibate = this
                    }
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:celibate }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:celibate_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_active_filter_list target = flag:celibate_negative }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}
