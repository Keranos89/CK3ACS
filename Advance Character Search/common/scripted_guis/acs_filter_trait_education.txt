acs_filter_trait_education_diplomacy_1 = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_filter_trait_education_general_list
            flag:education_diplomacy_1 = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_filter_trait_education_general_list
                        OR = {
                            flag:education_diplomacy_1 = this
                            flag:education_diplomacy_1_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_filter_trait_education_general_list target = flag:education_diplomacy_1 }
            
            set_local_variable = { name = acs_temp value = 0 }
            every_in_global_list = {
                variable = acs_filter_trait_education_general_list
                prev = { change_local_variable = { name = acs_temp add = 1 } }
            }
            if = {
                limit = {
                    local_var:acs_temp = 1
                }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:acs_filter_trait_education_general }
            }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_filter_trait_education_general_list
                        flag:education_diplomacy_1 = this
                    }
                }
                remove_list_global_variable = { name = acs_filter_trait_education_general_list target = flag:education_diplomacy_1 }
                add_to_global_variable_list = { name = acs_filter_trait_education_general_list target = flag:education_diplomacy_1_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_filter_trait_education_general_list target = flag:education_diplomacy_1_negative }
            
            set_local_variable = { name = acs_temp value = 0 }
            every_in_global_list = {
                variable = acs_filter_trait_education_general_list
                prev = { change_local_variable = { name = acs_temp add = 1 } }
