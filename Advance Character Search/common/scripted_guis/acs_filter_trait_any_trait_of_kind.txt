acs_filter_trait_any_sinful_negative = {
    is_shown = {
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:acs_filter_trait_any_sinful_negative = this
        }
    }
}

acs_filter_trait_any_sinful = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:acs_filter_trait_any_sinful = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        OR = {
                            flag:acs_filter_trait_any_sinful = this
                            flag:acs_filter_trait_any_sinful_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_active_filter_list target = flag:acs_filter_trait_any_sinful }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        flag:acs_filter_trait_any_sinful = this
                    }
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:acs_filter_trait_any_sinful }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:acs_filter_trait_any_sinful_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_active_filter_list target = flag:acs_filter_trait_any_sinful_negative }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}
