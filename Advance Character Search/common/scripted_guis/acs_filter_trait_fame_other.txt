    is_shown = {
        any_in_global_list = {
            variable = acs_filter_trait_fame_born_status_list
            flag:bastard_negative = this
        }
    }
}

acs_filter_trait_fame_born_status_bastard = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_filter_trait_fame_born_status_list
            flag:bastard = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_filter_trait_fame_born_status_list
                        OR = {
                            flag:bastard = this
                            flag:bastard_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_filter_trait_fame_born_status_list target = flag:bastard }
            
            set_local_variable = { name = acs_temp value = 0 }
            every_in_global_list = {
                variable = acs_filter_trait_fame_born_status_list
                prev = { change_local_variable = { name = acs_temp add = 1 } }
            }
            if = {
                limit = {
                    local_var:acs_temp = 1
                }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:acs_filter_trait_fame_born_status }
            }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_filter_trait_fame_born_status_list
                        flag:bastard = this
                    }
                }
                remove_list_global_variable = { name = acs_filter_trait_fame_born_status_list target = flag:bastard }
                add_to_global_variable_list = { name = acs_filter_trait_fame_born_status_list target = flag:bastard_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_filter_trait_fame_born_status_list target = flag:bastard_negative }
            
            set_local_variable = { name = acs_temp value = 0 }
            every_in_global_list = {
                variable = acs_filter_trait_fame_born_status_list
                prev = { change_local_variable = { name = acs_temp add = 1 } }
            }
            if = {
                limit = {
                    local_var:acs_temp = 0
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:acs_filter_trait_fame_born_status }
            }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}
