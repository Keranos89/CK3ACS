acs_filter_trait_stress_journaller_negative = {
    is_shown = {
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:journaller_negative = this
        }
    }
}

acs_filter_trait_stress_journaller = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:journaller = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        OR = {
                            flag:journaller = this
                            flag:journaller_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_active_filter_list target = flag:journaller }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        flag:journaller = this
                    }
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:journaller }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:journaller_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_active_filter_list target = flag:journaller_negative }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}
