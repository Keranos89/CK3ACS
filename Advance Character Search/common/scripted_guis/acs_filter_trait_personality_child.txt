acs_filter_trait_child_rowdy_negative = {
    is_shown = {
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:rowdy_negative = this
        }
    }
}

acs_filter_trait_child_rowdy = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:rowdy = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        OR = {
                            flag:rowdy = this
                            flag:rowdy_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_active_filter_list target = flag:rowdy }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        flag:rowdy = this
                    }
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:rowdy }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:rowdy_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_active_filter_list target = flag:rowdy_negative }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}
