acs_filter_trait_child_rowdy = {
 
    is_shown = { 
        global_var:acs_filter_trait_child_rowdy = yes
    }  
        
    effect = {
        if = {
            limit = { NOT = {  has_global_variable = acs_filter_trait_child_rowdy } }
            set_global_variable = {  name = acs_filter_trait_child_rowdy value = yes }
        }
        else = {
            if = {
                limit = { global_var:acs_filter_trait_child_rowdy = yes }
                set_global_variable = {  name = acs_filter_trait_child_rowdy value = no }
            }
            else = {
                remove_global_variable = acs_filter_trait_child_rowdy
            }

        }
        acs_auto_apply_sorting_and_filters = yes
    }
}
