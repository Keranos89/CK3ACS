acs_filter_trait_commander_logistician = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:logistician = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        OR = {
                            flag:logistician = this
                            flag:logistician_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_active_filter_list target = flag:logistician }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        flag:logistician = this
                    }
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:logistician }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:logistician_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_active_filter_list target = flag:logistician_negative }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}

        any_in_global_list = {
            variable = acs_active_filter_list
            flag:rough_terrain_expert = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        OR = {
                            flag:rough_terrain_expert = this
                            flag:rough_terrain_expert_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_active_filter_list target = flag:rough_terrain_expert }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        flag:rough_terrain_expert = this
                    }
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:rough_terrain_expert }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:rough_terrain_expert_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_active_filter_list target = flag:rough_terrain_expert_negative }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}
