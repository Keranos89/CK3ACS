acs_filter_trait_religius_saoshyant_negative = {
    is_shown = {
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:saoshyant_negative = this
        }
    }
}

acs_filter_trait_religius_saoshyant = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:saoshyant = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        OR = {
                            flag:saoshyant = this
                            flag:saoshyant_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_active_filter_list target = flag:saoshyant }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        flag:saoshyant = this
                    }
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:saoshyant }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:saoshyant_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_active_filter_list target = flag:saoshyant_negative }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}
