acs_filter_trait_health_gout_ridden_negative = {
    is_shown = {
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:gout_ridden_negative = this
        }
    }
}

acs_filter_trait_health_gout_ridden = {
 
    is_shown = { 
        any_in_global_list = {
            variable = acs_active_filter_list
            flag:gout_ridden = this
        }
    }  
        
    effect = {
        acs_save_undo_0_filters = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        OR = {
                            flag:gout_ridden = this
                            flag:gout_ridden_negative = this
                        }
                    } 
                } 
            }
            add_to_global_variable_list = { name = acs_active_filter_list target = flag:gout_ridden }
        }
        else = {
            if = {
                limit = { 
                    any_in_global_list = {
                        variable = acs_active_filter_list
                        flag:gout_ridden = this
                    }
                }
                remove_list_global_variable = { name = acs_active_filter_list target = flag:gout_ridden }
                add_to_global_variable_list = { name = acs_active_filter_list target = flag:gout_ridden_negative }
            }
            else = {
                remove_list_global_variable = { name = acs_active_filter_list target = flag:gout_ridden_negative }
            }
        }
       
        acs_auto_apply_sorting_and_filters = yes
    }
}
