acs_filter_simple_checkbox = {
    scope = province
    saved_scopes = {
		major_digit
        minor_digit
	}
    is_shown = { 
        any_in_global_list = {
            variable = acs_active_filter_list
            this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = var:index add = -1 }
        }
    }     
    effect = {
        set_local_variable = { name = acs_start value = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index } }
        acs_simple_checkbox = { LIST = acs_active_filter_list START = local_var:acs_start }
    }
}


#  0 = 0 1
#  1 = 0 2
#  2 = 1 0
#  3 = 1 1
#  4 = 1 2
#  5 = 2 0
#  6 = 2 1
#  7 = 2 2
acs_filter_2_group_checkbox = {
    scope = province
    saved_scopes = {
		major_digit
        minor_digit
        element_in_group
	}
    is_shown = { 
        trigger_if = {
            limit = { 
                scope:element_in_group.var:index = 0
                var:index = 1
            }
            any_in_global_list = {
                variable = acs_active_filter_list
                OR = {
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 2 }
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 3 }
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 4 }
                }
            }
        }
        trigger_else_if = {
            limit = { 
                scope:element_in_group.var:index = 0
                var:index = 2
            }
            any_in_global_list = {
                variable = acs_active_filter_list
                OR = {
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 5 }
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 6 }
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 7 }

                }
            }
        }
        trigger_else_if = {
            limit = { 
                scope:element_in_group.var:index = 1
                var:index = 1
            }
            any_in_global_list = {
                variable = acs_active_filter_list
                OR = {
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index }
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 3 }
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 6 }
                }
            }
        }
        trigger_else = {
            any_in_global_list = {
                variable = acs_active_filter_list
                OR = {
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 1 }
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 4 }
                    this = { value = scope:major_digit.var:index multiply = 40 add = scope:minor_digit.var:index add = 7 }
                }
            }
