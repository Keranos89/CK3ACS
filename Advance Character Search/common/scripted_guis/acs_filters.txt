acs_filter_simple_checkbox = {
    scope = province
    saved_scopes = {
        major_digit
        minor_digit
    }
    is_shown = {
        dummy_male = {
            any_in_list = {
                variable = acs_active_filter_list
                save_temporary_scope_value_as = {name = offset value = -1 }
                this = acs_major_minor_index_offset
            }
        }
    }     
    effect = {
        acs_simple_checkbox = { LIST = acs_active_filter_list START = acs_major_minor }
        acs_auto_apply_sorting_and_filters = yes
    }
}


#  0 = 0 1
#  1 = 0 2
#  2 = 1 0
#  3 = 1 1
#  4 = 1 2
#  5 = 2 0
#  6 = 2 1
#  7 = 2 2
acs_filter_2_group_checkbox = {
    scope = province
    saved_scopes = {
        major_digit
        minor_digit
        element_in_group
    }
    is_shown = {
        dummy_male = {
            trigger_if = {
                limit = { 
                    scope:element_in_group.var:index = 0
                    root.var:index = 1
                }
                any_in_list = {
                    variable = acs_active_filter_list
                    OR = {
                        save_temporary_scope_value_as = {name = offset value = 2 }
                        this = acs_major_minor_offset
                        save_temporary_scope_value_as = {name = offset value = 3 }
                        this = acs_major_minor_offset
                        save_temporary_scope_value_as = {name = offset value = 4 }
                        this = acs_major_minor_offset
                    }
                }
            }
            trigger_else_if = {
                limit = { 
                    scope:element_in_group.var:index = 0
                    root.var:index = 2
                }
                any_in_list = {
                    variable = acs_active_filter_list
                    OR = {
                        save_temporary_scope_value_as = {name = offset value = 5 }
                        this = acs_major_minor_offset
                        save_temporary_scope_value_as = {name = offset value = 6 }
                        this = acs_major_minor_offset
                        save_temporary_scope_value_as = {name = offset value = 7 }
                        this = acs_major_minor_offset
                    }
                }
            }
            trigger_else_if = {
                limit = { 
                    scope:element_in_group.var:index = 1
                    root.var:index = 1
                }
                any_in_list = {
                    variable = acs_active_filter_list
                    OR = {
                        this = acs_major_minor
                        save_temporary_scope_value_as = {name = offset value = 3 }
                        this = acs_major_minor_offset
                        save_temporary_scope_value_as = {name = offset value = 6 }
                        this = acs_major_minor_offset
                    }
                }
            }
            trigger_else = {
                any_in_list = {
                    variable = acs_active_filter_list
                    OR = {
                        save_temporary_scope_value_as = {name = offset value = 1 }
                        this = acs_major_minor_offset
                        save_temporary_scope_value_as = {name = offset value = 4 }
                        this = acs_major_minor_offset
                        save_temporary_scope_value_as = {name = offset value = 7 }
                        this = acs_major_minor_offset
                    }
                }
            }
        }

    }  
        
    effect = {
        dummy_male = {
            acs_save_undo_0_filters = yes
            if = {
                limit = {
                    NOT = {
                        any_in_list = {
                            variable = acs_active_filter_list
